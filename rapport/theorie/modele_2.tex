\section{Introduction}
D'après l'article "JGR Solid Earth", on a l'expression de la pression fluide en fonction du temps t à une dsitance r du point d'injection (où $q$ est le débit volumique, $k$ la perméabilité, $\eta$ la viscosité dynamique et $\rho}_0$ la masse volumique du fluide) :

\begin{equation}
P(t)
= \frac{q }{4\pi{\rho}_0 r}\frac{\eta}{k}\operatorname{erfc}(\frac{r}{2\sqrt{ct}})= P_{\infty}\operatorname{erfc}(\frac{r}{2\sqrt{ct}})
\end{equation}

La pression de fluide "soulage" la contrainte normale et on a:
\begin{equation}
\eta v 
= -k\left( l^{*} + \delta - v_p t \right)
  - f(t)(\sigma -P(t))
\end{equation}

On dérive par rapport au temps :
\begin{equation}
\eta \frac{dv}{dt}
= -k (v - v_p)
  - \left(\frac{df}{dt}(\sigma -P(t))-\frac{dP}{dt}f(t)\right)
\end{equation}

On multiplie par $\frac{d_c}{v_p b \sigma}$ (idem que le cas sans pression de fluide) :

\begin{equation}
\left(\bar{\eta}+\frac{\alpha}{\bar{v}}(1-\frac{P(t)}{\sigma})\right) \frac{d\bar{v}}{d\bar{t}}
=
-\kappa (\bar{v} - 1) + (\bar{v}-\frac{1}{\bar{\theta}})(1-\frac{P(t)}{\sigma})
  - \left(f_0+a\ln{\bar{(v)}}+b\ln{\bar{(\theta)}}\right)\frac{d_c}{v_p b \sigma}\frac{dP}{dt}
\end{equation}



Or, on a:

\begin{equation}
\frac{P(t)}{\sigma}=\bar{P_{\infty}}\operatorname{erfc}(\frac{\bar{r}}{2\sqrt{\bar{c}\bar{t}}})
\end{equation}

\begin{equation} 
\frac{d_c}{v_p b \sigma}\frac{dP}{dt} =\frac{1}{b}\frac{1}{2\sqrt{\pi}}\frac{\bar{P_{\infty}}}{\bar{t}}\frac{\bar{r}}{\sqrt{\bar{c}\bar{t}}}
e^{-\left(\frac{\bar{r}}{2\sqrt{\bar{c}\bar{t}}}\right)^2}
\end{equation}
Les variables adimensionnées étant:
$$
\bar{r} = \frac{r}{L^{\star}}=\frac{r}{\frac{\mu d_c}{b \sigma}}, \qquad 
\bar{c} = c\frac{b^2\sigma^2}{v_p \mu^2 d_c}, \qquad 
\bar{P_{\infty}}=\frac{P_{\infty}}{\sigma}
$$


\begin{equation}
\fbox{\[
\displaystyle \frac{-\kappa(\bar{v} - 1)
  +(\bar{v}
  - \frac{1}{\bar{\theta}})(1-\bar{P_{\infty}}\operatorname{erfc}(\frac{\bar{r}}{2\sqrt{\bar{c}\bar{t}}}))-\left(\frac{f_0}{b}+\alpha\ln{\bar{(v)}}+\ln{\bar{(\theta)}}\right)\frac{\bar{P_{\infty}}}{\bar{t}}\frac{\bar{r}}{2\sqrt{\pi\bar{c}\bar{t}}}e^{-\left(\frac{\bar{r}}{2\sqrt{\bar{c}\bar{t}}}\right)^2}}{\bar{\eta}\bar{v}+\alpha(1-\bar{P_{\infty}}\operatorname{erfc}(\frac{\bar{r}}{2\sqrt{\bar{c}\bar{t}}}))}d\bar{t} = \frac{d\bar{v}}{\bar{v}} \]} 
\end{equation}
